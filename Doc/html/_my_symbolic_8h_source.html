99  // 检查括号平衡
100  bool IsBracketBalance(std::stringstream& input)
101  {
102  std::string save;
103  save = input.str();
104  std::string temps;
105  std::stack<bool> bracketStack;
106  while (input >> temps)
107  {
108  if (temps == "(")
109  {
110  bracketStack.push(true);
111  }
112  else if (temps == ")")
113  {
114  bracketStack.pop();
115  }
116  else { ; }
117  }
118  input.clear();
119  input.str("");
120  input << save;
121  return bracketStack.empty();
122  }

220  // 由中缀表达式构造表达式树(允许采用括号)
221  void BuildFromInfixExpr(std::stringstream& input)
222  {
223  // 首先检查括号平衡
224  if (!IsBracketBalance(input))
225  {
226  return;
227  }
228  else { ; }
229 
230  // 清除原先的表达式树
231  ClearTree();
232  // 用于构造表达式树的操作符栈(包含了括号操作符)
233  std::stack<std::string> opStack;
234  // 用于构造表达式树的操作数栈
235  std::stack<TreeNode<ET>* > dataStack;
236  // 当前的操作数/操作符
237  std::string curString;
238  // 循环遍历
239  while (input >> curString)
240  {
241  if (operators.find(curString) != operators.end() || curString == "(" || curString == ")")
242  {
243  // 说明是操作符
244  if (curString == "(")
245  {
246  // 左括号直接入栈
247  opStack.push(curString);
248  }
249  else if (curString == ")")
250  {
251  // 右括号和上一个左括号对应
252  // 为了防止冗赘括号，之前已检查强制保证括号平衡
253  while (!dataStack.empty() && opStack.top() != "(")
254  {
255  // 从操作数栈取出两个操作数
256  TreeNode<ET>* secondOpd = dataStack.top();
257  dataStack.pop();
258  TreeNode<ET>* firstOpd = dataStack.top();
259  dataStack.pop();
260  // 将操作数和操作符组成一个新结点存入栈中
261  dataStack.push(new TreeNode<ET>(opStack.top(), nullptr, firstOpd, secondOpd));
262  opStack.pop();
263  }
264  // 将左括号出栈
265  opStack.pop();
266  }
267  else
268  {
269  //如果栈顶操作符优先级高于读入操作符优先级，则表名应该先计算栈顶操作符
270  while (!opStack.empty() && comparer(opStack.top(), curString))
271  {
272  TreeNode<ET>* secondOpd = dataStack.top();
273  dataStack.pop();
274  TreeNode<ET>* firstOpd = dataStack.top();
275  // 从操作数栈取出两个操作数
276  dataStack.pop();
277  // 将操作数和操作符组成一个新结点存入栈中
278  dataStack.push(new TreeNode<ET>(opStack.top(), nullptr, firstOpd, secondOpd));
279  opStack.pop();
280  }
281  // 将读入操作符入栈
282  opStack.push(curString);
283  }
284  }
285  else
286  {
287  // 说明是操作数
288  auto element = data.at(curString);
289  dataStack.push(new TreeNode<ET>(curString, element));
290  }
291  }
292  curString = "";
293  // 单独处理根节点
294  while (!opStack.empty() && comparer(opStack.top(), curString))
295  {
296  TreeNode<ET>* secondOpd = dataStack.top();
297  dataStack.pop();
298  TreeNode<ET>* firstOpd = dataStack.top();
299  dataStack.pop();
300  dataStack.push(new TreeNode<ET>(opStack.top(), nullptr, firstOpd, secondOpd));
301  opStack.pop();
302  }
303  // 将唯一的剩余的节点设为根节点
304  root = dataStack.top();
305  dataStack.pop();
306  }

308  template<typename TS>
309  void PrintToInfixExpr(TS& out)
310  {
311  //结点栈，遍历使用
312  std::stack<TreeNode<ET>* > nodeStack;
313  // 当前遍历节点初始化为表达式树根节点
314  TreeNode<ET>* pointer = root;
315  // 用于记录在哪些元素被输出之后要输出反括号
316  std::list<TreeNode<ET>* > nodeList;
317 
318  // lambda对象：判断是否需要添加左括号保证运算顺序的正确性
319  auto shouldPrintLeftBracket = [&](bool isLeft)->bool
320  {
321  if (nodeStack.empty())
322  return false;
323  if (pointer == nullptr)
324  return false;
325  auto a = nodeStack.top()->name;
326  auto b = pointer->name;
327  //如果是数字，则不用打括号
328  if (operators.find(b) == operators.end())
329  return false;
330  if (isLeft)
331  {
332  // 如果pointer是左结点
333  if (this->comparer(b, a))
334  {
335  return false;
336  }
337  else
338  {
339  return true;
340  }
341  }
342  else
343  {
344  // 如果pointer是右结点
345  if (this->comparer(a, b))
346  {
347  return true;
348  }
349  else
350  {
351  return false;
352  }
353  }
354  return false;
355  };
356 
357  while (!nodeStack.empty() || pointer != nullptr)
358  {
359  // 一直向左子结点走，找到左子结点时，经过的结点已全部入栈
360  while (pointer != nullptr)
361  {
362  // 如果应该添加左括号,为左子结点的情况下
363  if (shouldPrintLeftBracket(true))
364  {
365  // 找到应该在输出哪个结点后输出右括号
366  auto temp = pointer->rightNode;
367  while (temp->rightNode != nullptr)
368  {
369  temp = temp->rightNode;
370  }
371  // 待插入右括号的节点位置入栈
372  nodeList.push_back(temp);
373  out << " ( ";
374  }
375  nodeStack.push(pointer);
376  pointer = pointer->leftNode;
377  }
378  // 输出结点
379  out << " " + nodeStack.top()->name + " ";
380  auto it = std::find(nodeList.begin(), nodeList.end(), nodeStack.top());
381  // 若栈顶结点(即当前输出结点)为前面记录的应该输出右括号的结点时，输出右括号，可能有多个右括号需要输出，所以循环
382  while (it != nodeList.end())
383  {
384  out << " ) ";
385  nodeList.erase(it);
386  it = std::find(nodeList.begin(), nodeList.end(), nodeStack.top());
387  }
388  // 所有左结点已经走完，走右节点
389  pointer = nodeStack.top()->rightNode;
390  // 如果应该添加左括号,为右子结点的情况下
391  if (shouldPrintLeftBracket(false))
392  {
393  auto temp = pointer->rightNode;
394  while (temp->rightNode != nullptr)
395  {
396  temp = temp->rightNode;
397  }
398  nodeList.push_back(temp);
399  out << " ( ";
400  }
401  nodeStack.pop();
402  }
403  out << std::endl;
404  }
