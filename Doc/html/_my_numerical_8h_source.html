42  // 基础数据结构：区间Interval(TX 只允许是整型/浮点类型)
43  template<typename TX, std::enable_if_t<std::is_floating_point_v<TX> || std::is_integral_v<TX>, int> = 0 >
44  class Interval
45  {
46  private:
47  // 左端点(true表示闭区间，false表示开区间)
48  std::pair<TX, bool> _begin;
49  // 右端点(true表示闭区间，false表示开区间)
50  std::pair<TX, bool> _end;
51  public:
52  // 返回区间左端点
53  std::pair<TX, bool> Begin() { return _begin; }
54  // 设置区间左端点
55  void Begin(TX x1, bool isClosed) { _begin = { x1, isClosed }; }
56  // 返回区间右端点
57  std::pair<TX, bool> End() { return _end; }
58  // 设置区间右端点
59  void End(TX x2, bool isClosed) { _end = { x2, isClosed }; }
60  // 判断是否在区间内
61  bool Contain(TX x)
62  {
63  bool result = true;
64  // 判断是否在左端点的右侧
65  _begin.second ?
66  result = result && x >= _begin.first : result = result && x > _begin.first;
67  // 判断是否在右端点左侧
68  _end.second ?
69  result = result && x <= _end.first : result = result && x < _end.first;
70  return result;
71  }
72  // 打印区间
73  std::string Print()
74  {
75  std::string result;
76  // 判断左端点开闭
77  _begin.second ?
78  result += "[" : result += "(";
79  // 添加区间
80  result += std::to_string(_begin.first) + ", " + std::to_string(_end.first);
81  // 判断右端点开闭
82  _end.second ?
83  result += "]" : result += ")";
84  return std::move(result);
85  }
86  public:
87  Interval()
88  {
89 
90  }

92  : _begin(x1, false), _end(x2, false)

104  // 基础数据结构：分段函数SegmentFunction
105  // 注意，不保证分段的顺序
106  template<typename TX, typename TY>
107  class SegmentFunction
108  {
109  private:
110  // 存放分段定义域 [a, b]
111  std::vector<Interval<TX> > _Segments;
112  // 存放分段函数 Fun: [a, b] -> R
113  std::vector<std::function<TY(TX)> > _Functions;
114  // 存放分段函数的描述
115  std::vector<std::string> _Description;
116  // 总分段数
117  unsigned int _NumSegment;
118  public:
119  // 返回时所在分段的Offset
120  size_t GetSegment(TX x)
121  {
122  size_t result = 0;
123  while (!_Segments[result].Contain(x))
124  {
125  result++;
126  }
127  return result;
128  }
129  // 添加新分段
130  void AddSegment(Interval<TX> inter, std::function<TY(TX)> fun, std::string des="")
131  {
132  _Segments.emplace_back(inter);
133  _Functions.emplace_back(fun);
134  _Description.emplace_back(des);
135  _NumSegment++;
136  }
137  void AddSegment(size_t offSet, Interval<TX> inter, std::function<TY(TX)> fun, std::string des = "")
138  {
139  if (offSet >= _NumSegment)
140  {
141  AddSegment(inter, fun, des);
142  }
143  else
144  {
145  _Segments[offSet] = inter;
146  _Functions[offSet] = fun;
147  _Description[offSet] = des;
148  }
149  }
150  // 清除分段函数
151  void Clear()
152  {
153  _Segments.clear();
154  _Functions.clear();
155  _Description.clear();
156  _NumSegment = 0;
157  }
158  // 重载operator()
159  // 警告，不检查定义域
160  TY operator() (TX x)
161  {
162  return _Functions[GetSegment(x)](x);
163  }
164  // 打印分段函数
165  std::string Print()
166  {
167  std::string result("Function:\n");
168  for (size_t seg = 0; seg < _NumSegment; seg++)
169  {
170  result += "\t" + _Description[seg] + " : " + _Segments[seg].Print() + "\n";
171  }
172  return std::move(result);
173  }
174  public:
175  // 构造函数
176  SegmentFunction()
177  {
178 
179  }
180  SegmentFunction(int numSegment)
181  : _NumSegment(numSegment > 0 ? numSegment : 0)
182  {
183  _Segments = std::vector<Interval<TX> >(_NumSegment);
184  _Functions = std::vector<std::function<TY(TX)> >(_NumSegment);
185  _Description = std::vector<std::string>(_NumSegment);
186  }
187  // 禁止列表初始化
188  SegmentFunction(std::initializer_list<std::pair<Interval<TX>, std::function<TY(TX)> > >) = delete;
189  };
